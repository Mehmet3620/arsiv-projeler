        document.addEventListener('DOMContentLoaded', () => {

            // Siyah Player için DOM Elementleri

            const volumeBarBlack = document.getElementById('volumeBarBlack');

            const radioWidgetBlue = document.getElementById('radioWidgetBlue');
            const toggleBtnWrapperBlue = document.getElementById('toggleBtnWrapperBlue');
            const statusTextBlue = document.getElementById('statusTextBlue');
            const volumeMuteIconContainerBlue = document.getElementById('volumeMuteIconBlue');
            const volumeIconBlue = document.getElementById('volumeIconBlue');
            const muteIconBlue = document.getElementById('muteIconBlue');
            const volumeSliderProgressBlue = document.getElementById('volumeSliderProgressBlue');
            const volumeBarBlue = document.getElementById('volumeBarBlue');
            
            // Turuncu Player için DOM Elementleri (YENİ)
            const audioOrange = document.getElementById('radio-audio-orange');
            const radioWidgetOrange = document.getElementById('radioWidgetOrange');
            const toggleBtnWrapperOrange = document.getElementById('toggleBtnWrapperOrange');
            const statusTextOrange = document.getElementById('statusTextOrange');
            const volumeMuteIconContainerOrange = document.getElementById('volumeMuteIconOrange');
            const volumeIconOrange = document.getElementById('volumeIconOrange');

            const volumeSliderProgressOrange = document.getElementById('volumeSliderProgressOrange');
            const volumeBarOrange = document.getElementById('volumeBarOrange');


            // Player durumlarını ve ayarlarını yöneten jenerik fonksiyon
            function initializePlayer(audioElement, widgetElement, toggleBtn, statusTxt, volumeMuteIconCont, volIcon, muteIc, volSliderProgress, volBar, themeName) {
                let isPlaying = false; 
                let elapsed = 0;
                let lastKnownVolume = 0.5; 
                let volumeChangeTimer = null;
                let isDragging = false;

                audioElement.volume = lastKnownVolume;
                updateVolumeUI(lastKnownVolume);

                function getVolumePercent(volume) {
                    return Math.round(volume * 100); 
                }
                
                function updateVolumeUI(volume) {
                    const percent = getVolumePercent(volume);
                    volBar.style.width = `${percent}%`;
                    
                    if (volume < 0.01) { 
                        muteIc.classList.add('active');
                        volIcon.classList.remove('active');
                    } else {
                        muteIc.classList.remove('active');
                        volIcon.classList.add('active');
                    }
                }
                
                function updateStatusText(messageType, currentVolumePercent = 0) {
                    if (!isPlaying) {
                        statusTxt.textContent = 'Player stopped.'; 
                        return;
                    }
                    
                    let streamText = `Playing GURBETTEYİM FM `;
                    let timeText = `${Math.floor(elapsed / 60).toString().padStart(2, '0')}:${(elapsed % 60).toString().padStart(2, '0')}`;
                    
                    if (messageType === 'volume') {
                         statusTxt.textContent = `Volume %${Math.round(currentVolumePercent)}`;
                    } else {
                         statusTxt.textContent = streamText + timeText;
                    }
                }

                function updatePlayerState(isManualUpdate = false) {
                    if (isPlaying) {
                        widgetElement.classList.add('is-playing');
                        if (!volumeChangeTimer) {
                            updateStatusText('stream');
                        }
                    } else {
                        widgetElement.classList.remove('is-playing');
                        statusTxt.textContent = 'Player stopped.'; 
                    }
                    if (isManualUpdate) {
                        updateVolumeUI(audioElement.volume);
                    }
                }
                
                toggleBtn.addEventListener('click', () => {
                    if (isPlaying) {
                        isPlaying = false; 
                        elapsed = 0;
                        updatePlayerState(true); 
                        
                        audioElement.pause(); 
                        audioElement.currentTime = 0; 
                        
                    } else {
                        isPlaying = true; 
                        updatePlayerState(true); 
                        
                        if (audioElement.volume < 0.01) {
                             audioElement.volume = lastKnownVolume > 0.01 ? lastKnownVolume : 0.5;
                             updateVolumeUI(audioElement.volume);
                        }
                        
                        audioElement.load(); 
                        audioElement.play().catch(error => {
                            console.error(`[${themeName} Player] Oynatma hatası:`, error);
                            isPlaying = false;
                            updatePlayerState(true);
                            statusTxt.textContent = 'Oynatma başlatılamadı.';
                        });
                    }
                });

                volumeMuteIconCont.addEventListener('click', () => {
                    if (volumeChangeTimer) clearTimeout(volumeChangeTimer);
                    
                    if (audioElement.volume > 0.01) {
                        lastKnownVolume = audioElement.volume; 
                        audioElement.volume = 0;
                    } else {
                        audioElement.volume = lastKnownVolume > 0.01 ? lastKnownVolume : 0.5;
                    }
                    
                    updateVolumeUI(audioElement.volume);
                    updateStatusText('volume', getVolumePercent(audioElement.volume));
                    
                    volumeChangeTimer = setTimeout(() => {
                        volumeChangeTimer = null;
                        updateStatusText('stream');
                    }, 2000);
                });
                
                const startDrag = (e) => {
                    e.preventDefault();
                    isDragging = true;
                    document.addEventListener('mousemove', onDrag);
                    document.addEventListener('mouseup', stopDrag);
                    document.addEventListener('touchmove', onDrag);
                };

                const onDrag = (e) => {
                    if (!isDragging) return;
                    
                    if (volumeChangeTimer) clearTimeout(volumeChangeTimer);
                    
                    const rect = volSliderProgress.getBoundingClientRect();
                    let clientX = e.clientX || (e.touches ? e.touches[0].clientX : 0);
                    
                    let newVolume = (clientX - rect.left) / rect.width;
                    newVolume = Math.max(0, Math.min(1, newVolume)); 
                    
                    audioElement.volume = newVolume.toFixed(2);
                    lastKnownVolume = audioElement.volume;
                    
                    updateVolumeUI(audioElement.volume);
                    updateStatusText('volume', getVolumePercent(audioElement.volume)); 
                };

                const stopDrag = () => {
                    isDragging = false;
                    document.removeEventListener('mousemove', onDrag);
                    document.removeEventListener('mouseup', stopDrag);
                    document.removeEventListener('touchmove', onDrag);
                    
                    volumeChangeTimer = setTimeout(() => {
                        volumeChangeTimer = null;
                        updateStatusText('stream');
                    }, 2000);
                };

                volSliderProgress.addEventListener('mousedown', startDrag);
                volSliderProgress.addEventListener('touchstart', startDrag);
                
                updatePlayerState(true);

                setInterval(() => {
                    if (isPlaying && !volumeChangeTimer) {
                         elapsed++;
                         updateStatusText('stream');
                    } else if (!isPlaying) {
                        elapsed = 0;
                    }
                }, 1000); 
            }

            // Player'ları başlat
            initializePlayer(audioBlack, radioWidgetBlack, toggleBtnWrapperBlack, statusTextBlack, volumeMuteIconContainerBlack, volumeIconBlack, muteIconBlack, volumeSliderProgressBlack, volumeBarBlack, 'Black');
            initializePlayer(audioBlue, radioWidgetBlue, toggleBtnWrapperBlue, statusTextBlue, volumeMuteIconContainerBlue, volumeIconBlue, muteIconBlue, volumeSliderProgressBlue, volumeBarBlue, 'Blue');
            initializePlayer(audioOrange, radioWidgetOrange, toggleBtnWrapperOrange, statusTextOrange, volumeMuteIconContainerOrange, volumeIconOrange, muteIconOrange, volumeSliderProgressOrange, volumeBarOrange, 'Orange');
        });
