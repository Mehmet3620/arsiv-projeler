<script>
    const videos = [
        { id: "1i6nc8NU8Sg", title: "Ey Yaralı Kalbim", time: "6:49" },
        { id: "m9Qjp2DXIK8", title: "Zehra", time: "5:22" },
        { id: "10EDraKWGsE", title: "Bir Delilik Yaptım", time: "4:15" },
        { id: "5VQr0rHDYOs", title: "Sıladan Gurbete Mektup", time: "5:40" },
        { id: "xLiDhanQQCM", title: "Ne Zaman Bir Tren Sesi Duysam", time: "4:55" },
        { id: "7eTK0ydE4Rw", title: "Benim Babam", time: "6:10" },
        { id: "otliMe05djQ", title: "Sen Gülünce", time: "3:45" },
        { id: "L1IJ-kdSjbI", title: "Yüreğimdesin Her Daim", time: "5:05" },
        { id: "qsQXs_Kpjc0", title: "Ayırmayın Sevenleri", time: "4:30" },
        { id: "3368lowsx7g", title: "Ana Bizim Bacı Bizim Yar Bizim", time: "5:50" }
    ];

    const playlistContainer = document.getElementById('playlist');
    const mainPlayer = document.getElementById('mainPlayer');

    function loadPlaylist() {
        videos.forEach((video, index) => {
            const item = document.createElement('div');
            item.className = `video-item ${index === 0 ? 'active' : ''}`;
            item.onclick = function() { playVideo(video.id, item); };

            item.innerHTML = `
                <img src="https://img.youtube.com/vi/${video.id}/mqdefault.jpg" class="thumb">
                <div class="video-info">
                    <div class="title">${video.title}</div>
                    <div class="duration">${video.time}</div>
                </div>
            `;
            playlistContainer.appendChild(item);
        });
    }

    function playVideo(id, element) {
        mainPlayer.src = "https://www.youtube-nocookie.com/embed/" + id + "?autoplay=1&rel=0";
        document.querySelectorAll('.video-item').forEach(el => el.classList.remove('active'));
        element.classList.add('active');
    }

    window.onload = loadPlaylist;
</script>
