<script>
    const player = document.getElementById('mpc-player');
    const handle = document.getElementById('drag-handle');
    const media = document.getElementById('main-video');
    const timeDisplay = document.getElementById('time-display');
    const playBtn = document.getElementById('play-btn');
    const pFill = document.getElementById('p-fill');
    const plItems = document.getElementById('pl-items');
    const canvas = document.getElementById('visualizer-canvas');
    const ctx = canvas.getContext('2d');

    window.onload = () => { centerWindow(); };
    function centerWindow() {
        player.style.width = '640px'; player.style.height = '480px';
        player.style.left = (window.innerWidth - 640) / 2 + 'px';
        player.style.top = (window.innerHeight - 480) / 2 + 'px';
    }
    function maximizeWindow() { player.style.left='0'; player.style.top='0'; player.style.width='100vw'; player.style.height='100vh'; }
    function restoreWindow() { centerWindow(); }

    let isDragging = false, startX, startY;
    handle.onmousedown = (e) => { 
        if(e.target.closest('.window-controls')) return;
        isDragging = true; startX = e.clientX - player.offsetLeft; startY = e.clientY - player.offsetTop; 
        player.style.transition = 'none'; 
    };
    window.onmousemove = (e) => { if (!isDragging) return; player.style.left = (e.clientX - startX) + 'px'; player.style.top = (e.clientY - startY) + 'px'; };
    window.onmouseup = () => { isDragging = false; player.style.transition = 'width 0.2s, height 0.2s, top 0.2s, left 0.2s'; };

    let playlist = [], curIdx = 0, visualMode = 0, aCtx, analyser, dataArray;

    document.getElementById('file-input').onchange = (e) => {
        const files = Array.from(e.target.files);
        if (files.length === 0) return;
        
        const firstAddedIdx = playlist.length;
        playlist = [...playlist, ...files]; 
        renderPL();
        
        // Önce dosyayı yükle sonra zorla oynat
        loadTrack(firstAddedIdx);
    };

    function renderPL() {
        plItems.innerHTML = playlist.map((f, i) => 
            `<div class="playlist-item ${i===curIdx?'active':''}" onclick="loadTrack(${i})">
                <span class="playlist-name">${i+1}. ${f.name}</span>
                <span class="del-item" onclick="event.stopPropagation(); removeTrack(${i})">×</span>
            </div>`
        ).join('');
    }

    function formatTime(s) {
        if(isNaN(s) || !isFinite(s)) return "00:00";
        const m = Math.floor(s / 60); const sec = Math.floor(s % 60);
        return (m < 10 ? "0" + m : m) + ":" + (sec < 10 ? "0" + sec : sec);
    }

    media.ontimeupdate = () => {
        pFill.style.width = (media.currentTime / media.duration * 100) + '%';
        timeDisplay.innerText = formatTime(media.currentTime) + " / " + formatTime(media.duration);
    };

    media.onended = () => {
        if (curIdx < playlist.length - 1) loadTrack(curIdx + 1);
    };

    function loadTrack(i) {
        if(i < 0 || i >= playlist.length) return;
        curIdx = i; 
        const file = playlist[i]; 
        
        // Mevcut blob URL'ini temizle (hafıza yönetimi)
        if(media.src) URL.revokeObjectURL(media.src);
        media.src = URL.createObjectURL(file);
        
        if(file.type.startsWith('video')) { 
            media.style.display = 'block'; 
            canvas.style.display = 'none'; 
        } else { 
            media.style.display = 'none'; 
            canvas.style.display = 'block'; 
            if(!aCtx) initAudioContext();
        }

        media.load();
        media.play().then(() => {
            playBtn.innerHTML = '<i class="fas fa-pause"></i>';
            document.getElementById('display-area').classList.add('playing');
        }).catch(err => {
            console.log("Oynatma hatası:", err);
            // Kullanıcı etkileşimi gerekirse butonu play yap
            playBtn.innerHTML = '<i class="fas fa-play"></i>';
        });
        
        renderPL();
    }

    function initAudioContext() {
        try {
            aCtx = new (window.AudioContext || window.webkitAudioContext)();
            analyser = aCtx.createAnalyser();
            const src = aCtx.createMediaElementSource(media);
            src.connect(analyser); analyser.connect(aCtx.destination);
            analyser.fftSize = 256; dataArray = new Uint8Array(analyser.frequencyBinCount);
            draw();
        } catch(e) { console.warn("Audio Context başlatılamadı."); }
    }

    function draw() {
        requestAnimationFrame(draw); if(!analyser || canvas.style.display === 'none') return;
        analyser.getByteFrequencyData(dataArray);
        canvas.width = canvas.offsetWidth; canvas.height = canvas.offsetHeight;
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        const color = getComputedStyle(document.body).getPropertyValue('--bar-fill') || '#0055e5';
        ctx.fillStyle = color;

        if(visualMode === 0) {
            const bw = (canvas.width / dataArray.length) * 2.5;
            dataArray.forEach((v, i) => { ctx.fillRect(i * bw, canvas.height - (v * (canvas.height/255)), bw - 1, v * (canvas.height/255)); });
        } else if(visualMode === 1) {
            ctx.beginPath(); ctx.strokeStyle = color; ctx.lineWidth = 2;
            const slice = canvas.width / dataArray.length;
            dataArray.forEach((v, i) => { const y = (canvas.height/2) + (v - 128); if(i===0) ctx.moveTo(i*slice, y); else ctx.lineTo(i*slice, y); });
            ctx.stroke();
        } else if(visualMode === 2) {
            ctx.beginPath(); ctx.strokeStyle = color;
            const centerX = canvas.width/2, centerY = canvas.height/2;
            for(let i=0; i<dataArray.length; i++) {
                const r = (Math.PI*2/dataArray.length)*i, h = dataArray[i]*0.5;
                const x = centerX + Math.cos(r)*(50+h), y = centerY + Math.sin(r)*(50+h);
                if(i===0) ctx.moveTo(x,y); else ctx.lineTo(x,y);
            }
            ctx.closePath(); ctx.stroke();
        } else {
            const bw = (canvas.width / dataArray.length);
            dataArray.forEach((v, i) => { ctx.globalAlpha = 0.5; ctx.fillRect(i * bw, (canvas.height/2)-(v/2), bw - 1, v); });
            ctx.globalAlpha = 1;
        }
    }

    function removeTrack(i) { playlist.splice(i, 1); renderPL(); }
    function clearPlaylist() { playlist = []; media.pause(); media.src = ""; renderPL(); document.getElementById('display-area').classList.remove('playing'); }
    function setVisual(mode) { visualMode = mode; }
    function setSkin(s) { document.body.className = s; }
    
    playBtn.onclick = () => { 
        if(media.paused) { media.play(); playBtn.innerHTML = '<i class="fas fa-pause"></i>'; } 
        else { media.pause(); playBtn.innerHTML = '<i class="fas fa-play"></i>'; } 
    };
    
    document.getElementById('stop-btn').onclick = () => { media.pause(); media.currentTime = 0; playBtn.innerHTML = '<i class="fas fa-play"></i>'; };
    document.getElementById('next-btn').onclick = () => loadTrack(curIdx + 1);
    document.getElementById('prev-btn').onclick = () => loadTrack(curIdx - 1);
    document.getElementById('v-slider').oninput = (e) => media.volume = e.target.value;
    document.getElementById('p-bar').onclick = (e) => media.currentTime = (e.offsetX / e.currentTarget.offsetWidth) * media.duration;
    document.getElementById('toggle-playlist').onclick = () => { const p = document.getElementById('playlist-panel'); p.style.display = p.style.display === 'block' ? 'none' : 'block'; };
    function skip(s) { media.currentTime += s; }
    function saveToPC() {
        const blob = new Blob([document.documentElement.outerHTML], {type: 'text/html'});
        const a = document.createElement('a'); a.href = URL.createObjectURL(blob);
        a.download = 'MPC_Seyyah_Fixed.html'; a.click();
    }
    function toggleAbout() { const ab = document.getElementById('about-box'); ab.style.display = ab.style.display === 'block' ? 'none' : 'block'; }
</script>
