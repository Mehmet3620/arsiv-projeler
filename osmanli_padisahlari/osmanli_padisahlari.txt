        const padisahGrid = document.getElementById('padisah-resim-listesi');
        const kartOverlay = document.getElementById('bilgi-karti-overlay');
        const kartDetay = document.getElementById('card-detail');
        const kartKapatBtn = document.getElementById('card-close-btn');

        /**
         * Padişah resimlerini ve adlarını grid'e yükler.
         */
        function renderPadisahList() {
            padisahGrid.innerHTML = '';
            
            PADISAH_DATA.forEach(padisah => {
                const item = document.createElement('div');
                item.className = 'padisah-item';
                item.setAttribute('data-id', padisah.id);
                
                const img = document.createElement('img');
                img.src = padisah.resim;
                img.alt = padisah.ad;
                
                // Resim yüklenemezse yedek resim kullanma denemesi (Hotlink engeline karşı)
                img.onerror = function() {
                    this.style.filter = 'grayscale(100%)'; 
                    this.alt = "Resim yüklenemedi: " + padisah.ad;
                };

                const name = document.createElement('p');
                name.textContent = `${padisah.id}. ${padisah.ad}`; 
                
                item.appendChild(img);
                item.appendChild(name);
                
                // Tıklama olayını tanımla
                item.addEventListener('click', () => openPadisahCard(padisah.id));
                
                padisahGrid.appendChild(item);
            });
        }

        /**
         * Padişah bilgi kartını açar ve içeriği yükler.
         * @param {number} id - Tıklanan padişahın ID'si
         */
        function openPadisahCard(id) {
            const padisah = PADISAH_DATA.find(p => p.id === id);
            if (!padisah) return;

            // Kullanıcı kuralı: uzun yazıları ikiye bölmek ve bölüm aralarına fazladan paragraf etiketi koymak
            let formattedBilgi = padisah.bilgi.replace(/<p><\/p>/g, '<br><p></p>');

            kartDetay.innerHTML = `
                <img src="${padisah.resim}" alt="${padisah.ad}">
                <h3>${padisah.id}. ${padisah.ad}</h3>
                <p><strong>Saltanat Süresi:</strong> ${padisah.saltanat}</p>
                <p>${formattedBilgi}</p>
            `;
            
            // Kart açıldığında resim yine yüklenemezse alternatif metni göster
            kartDetay.querySelector('img').onerror = function() {
                 this.src = 'Hata: Resim Yüklenemedi';
                 this.style.filter = 'grayscale(100%)'; 
            };


            kartOverlay.style.display = 'flex';
            document.body.style.overflow = 'hidden'; // Sayfanın kaymasını engelle
        }

        /**
         * Padişah bilgi kartını kapatır.
         */
        function closePadisahCard() {
            kartOverlay.style.display = 'none';
            document.body.style.overflow = 'auto';
        }

        // Olay dinleyicileri
        kartKapatBtn.addEventListener('click', closePadisahCard);
        kartOverlay.addEventListener('click', (e) => {
            // Kartın içeriğine değil, overlay'a tıklanırsa kapat
            if (e.target === kartOverlay) {
                closePadisahCard();
            }
        });

        // Sayfa yüklendiğinde listeyi render et
        document.addEventListener('DOMContentLoaded', () => {
            renderPadisahList();
        });
    </script>
