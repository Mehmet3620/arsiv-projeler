    function getHicriDate(date) {
        try { return new Intl.DateTimeFormat('tr-TR-u-ca-islamic', { day: 'numeric', month: 'long', year: 'numeric' }).format(date); } catch (e) { return "Hata"; }
    }
    
    function getDayOfYear(now) {
        const start = new Date(now.getFullYear(), 0, 0);
        const diff = now - start;
        const oneDay = 1000 * 60 * 60 * 24;
        return Math.floor(diff / oneDay);
    }
    
    function splitLongText(text) {
        // [2025-10-10] Uzun yazıları ikiye bölme ve paragraf ekleme kuralı uygulanmaktadır.
        const minLength = 50; 
        let paragraphs = [];
        let parts = text.split('(');
        let mainText = parts[0].trim();
        let reference = parts.length > 1 ? ' (' + parts.slice(1).join('(') : '';
        if (mainText.length > minLength) {
            let bestSplitIndex = -1;
            let searchIndex = mainText.indexOf('.');
            if (searchIndex === -1) { searchIndex = mainText.indexOf(','); }
            if (searchIndex !== -1) {
                let spaceAfterSplit = mainText.indexOf(' ', searchIndex + 1);
                bestSplitIndex = spaceAfterSplit !== -1 ? spaceAfterSplit : searchIndex;
            }
            if (bestSplitIndex !== -1 && bestSplitIndex > mainText.length * 0.3 && bestSplitIndex < mainText.length * 0.7) { 
                 paragraphs.push(mainText.substring(0, bestSplitIndex).trim());
                 paragraphs.push(mainText.substring(bestSplitIndex).trim() + reference);
            } else { paragraphs.push(mainText + reference); }
        } else { paragraphs.push(mainText + reference); }
        if (paragraphs.length === 2 && paragraphs[0].length > 0 && paragraphs[1].length > 0) {
            return `<p>${paragraphs[0]}</p><p>${paragraphs[1]}</p>`;
        }
        return `<p>${text}</p>`;
    }

    }

        try {
            const now = new Date();

            const monthName = TurkishMonths[now.getMonth()].toUpperCase();
            const dayOfMonth = now.getDate();
            
            calendarDayInfoEl.textContent = `${dayOfMonth} ${monthName} ${dayName}`;
            currentYearEl.textContent = now.getFullYear();

            miladiDateEl.textContent = `${dayOfMonth}.${now.getMonth() + 1}.${now.getFullYear()}`;
            hicriDateEl.textContent = getHicriDate(now);
            
            // 3. ZAMANI BAŞLAT (Her saniye çalışır)
            updateTime(); 
            setInterval(updateTime, 1000);

        } catch (error) {
            console.error("Pano başlatılırken kritik bir hata oluştu:", error);
        }
