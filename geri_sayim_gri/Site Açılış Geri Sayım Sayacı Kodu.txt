    <script>
        // Sitenin açılış tarihi: 1 Ocak 2026, Öğlen 12:00
        // YIL, AY(0-11), GÜN, SAAT, DAKİKA, SANİYE
        const siteLaunchDate = new Date(2026, 0, 1, 12, 0, 0).getTime(); 

        const siteCountdownFunction = setInterval(function() {
            const now = new Date().getTime();
            const distance = siteLaunchDate - now;

            // Hesaplamalar
            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            // HTML'deki yerlere yazma
            document.getElementById("site-days").innerHTML = days;
            document.getElementById("site-hours").innerHTML = hours;
            document.getElementById("site-minutes").innerHTML = minutes;
            document.getElementById("site-seconds").innerHTML = seconds;

            // Eğer geri sayım biterse
            if (distance < 0) {
                clearInterval(siteCountdownFunction); 
                document.getElementById("countdown-container-inner-site").style.display = "none"; 
                document.getElementById("site-status-message").style.display = "block"; 
            }
        }, 1000); 
    </script>
