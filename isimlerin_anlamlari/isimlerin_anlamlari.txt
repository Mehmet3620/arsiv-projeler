            </p>

            <h4>Kullanılmaması Gereken İsim Türleri</h4>
            <ul>
                <li>Allah'ın zâtına has isimler ("Allah" ve "Rahman"). Bu isimler sadece "Abd" (kul) kelimesi eklenerek ("Abdullah", "Abdurrahman") şeklinde kullanılabilir.</li>
                <li>Sadece Allah'a ait vasıfları ifade eden isimler ("Halık", "Samed", "Rezzak" vb.).</li>
            </ul>

<script>
    // TÜM İSİM LİSTELERİNİN ID'LERİ
    const nameTabIds = ['kizIsimleri1', 'kizIsimleri2', 'erkekIsimleri1', 'erkekIsimleri2'];

    // 1. SEKME İŞLEVİ
    // isFromFilter parametresi, bu çağrının filtreleme fonksiyonundan gelip gelmediğini kontrol eder.
    function openTab(tabId, isFromFilter = false) {
        // 1. Tüm içerikleri gizle ve başlıkların aktifliğini kaldır
        document.querySelectorAll('.tab-content').forEach(content => {
            content.style.display = 'none';
            content.classList.remove('active');
        });
        document.querySelectorAll('.tab-header').forEach(header => {
            header.classList.remove('active');
        });

        // 2. İlgili sekmeyi göster ve başlığı aktif yap
        const activeContent = document.getElementById(tabId);
        const activeHeader = document.querySelector(`.tab-header[onclick*='${tabId}']`);
        const searchContainer = document.getElementById('searchContainer');
        
        if (activeContent) {
            activeContent.style.display = 'block';
            activeContent.classList.add('active');
        }
        if (activeHeader) {
            activeHeader.classList.add('active');
        }

        // 3. Arama kutusu ve filtrelemeyi yönet
        if (tabId === 'aciklama') {
            searchContainer.style.display = 'none';
            document.getElementById('globalNoResult').style.display = 'none';
            resetSearch(); // Aramayı sıfırla
        } else {
            searchContainer.style.display = 'block';
            if (!isFromFilter) { 
                // Eğer sekme manuel olarak değiştirilmişse, mevcut filtreyi yeniden uygula
                filterIsimler(); 
            }
        }

        closeAllAccordions();
    }

    // 2. AKORDİYON İŞLEVİ
    function toggleAccordion(header) {
        const content = header.nextElementSibling;
        const icon = header.querySelector('i');
        
        // Sadece tek bir akordiyonun açık olmasını sağlamak için diğerlerini kapat
        document.querySelectorAll('.isim-header').forEach(h => {
            if (h !== header) {
                h.nextElementSibling.style.display = 'none';
                h.querySelector('i').classList.replace('fa-chevron-up', 'fa-chevron-down');
            }
        });

        // Tıklanan akordiyonu aç/kapat
        if (content.style.display === "block") {
            content.style.display = "none";
            icon.classList.replace('fa-chevron-up', 'fa-chevron-down');
        } else {
            content.style.display = "block";
            icon.classList.replace('fa-chevron-down', 'fa-chevron-up');
        }
    }

    function closeAllAccordions() {
        document.querySelectorAll('.isim-content').forEach(content => {
            content.style.display = 'none';
        });
        document.querySelectorAll('.isim-header i').forEach(icon => {
            icon.classList.replace('fa-chevron-up', 'fa-chevron-down');
        });
    }

    // 3. KRİTİK ARAMA İŞLEVİ (GLOBAL TARAMA VE OTOMATİK SEKME GEÇİŞİ)
    function filterIsimler() {
        const input = document.getElementById('isimSearch');
        const filter = input.value.toLocaleUpperCase('tr').trim(); 
        const globalNoResult = document.getElementById('globalNoResult');
        
        // Arama kutusu boşsa, tüm isimleri göster ve işlemi sonlandır
        if (filter.length === 0) {
            document.querySelectorAll('.isim-item').forEach(item => item.style.display = "");
            globalNoResult.style.display = 'none';
            return;
        }

        closeAllAccordions();
        let firstMatchingTabId = null;
        let globalMatchFound = false;

        // 1. Tüm isim listelerini (4 sekme) tarayarak filtreleme yap ve ilk eşleşeni bul
        nameTabIds.forEach(tabId => {
            const listContainer = document.getElementById(tabId);
            if (!listContainer) return; // Liste yoksa atla
            
            listContainer.querySelectorAll('.isim-item').forEach(item => {
                const nameText = item.querySelector('.name-text').textContent;
                const nameTextUpper = nameText.toLocaleUpperCase('tr');
                
                // İsim arama metniyle BAŞLIYORSA
                const isMatch = nameTextUpper.indexOf(filter) === 0;

                if (isMatch) {
                    item.style.display = ""; // Öğeyi göster
                    globalMatchFound = true;
                    if (!firstMatchingTabId) {
                        firstMatchingTabId = tabId; // İlk eşleşen sekmeyi kaydet
                    }
                } else {
                    item.style.display = "none"; // Öğeyi gizle
                }
            });
        });

        // 2. Global "Sonuç Yok" mesajını yönet
        if (!globalMatchFound) {
            globalNoResult.style.display = "block";
        } else {
            globalNoResult.style.display = "none";
            
            // 3. Eşleşme bulunduysa, sekmeyi otomatik olarak değiştir
            if (firstMatchingTabId) {
                const activeTabId = document.querySelector('.tab-content.active').id;
                
                // Eğer aktif sekme, ilk eşleşen sekme değilse, geçiş yap
                if (activeTabId !== firstMatchingTabId) {
                    // openTab fonksiyonunu çağırırken 'true' (isFromFilter) göndererek 
                    // sonsuz döngüyü ve gereksiz yeniden filtrelemeyi engelleriz.
                    openTab(firstMatchingTabId, true);
                }
            }
        }
    }

    // Aramayı ve sonuçları temizle
    function resetSearch() {
        document.getElementById('isimSearch').value = '';
        document.getElementById('globalNoResult').style.display = 'none';
        
        // Açıklama sekmesine geçiş yapıldığında tüm isimleri tekrar görünür yap
        document.querySelectorAll('.isim-item').forEach(item => item.style.display = "");
    }

    // Sayfa yüklendiğinde varsayılan sekmeyi aç
    document.addEventListener("DOMContentLoaded", () => {
        openTab('aciklama'); // Varsayılan olarak Açıklama sekmesini aç
    });
</script>
