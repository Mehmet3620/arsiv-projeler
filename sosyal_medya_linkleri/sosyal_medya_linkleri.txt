<script>
    function gosterSekme(icerikId, butonId) {
        var icerikler = document.querySelectorAll('.tab-icerik');
        icerikler.forEach(function(icerik) {
            icerik.style.display = 'none';
        });

        var butonlar = document.querySelectorAll('.tab-butonlari button');
        butonlar.forEach(function(buton) {
            buton.classList.remove('active');
        });

        var aktifIcerik = document.getElementById(icerikId);
        if (aktifIcerik) {
            aktifIcerik.style.display = 'block';
        }

        var aktifButon = document.getElementById(butonId);
        if (aktifButon) {
            aktifButon.classList.add('active');
        }
    }

    window.onload = function() {
        gosterSekme('konuA', 'btnA');
    };
</script>
